# Given a list of unsorted integers and a target value:
# - Return true if 2 values at distinct indicies sum up to target
# - Else return false
# Example:
# [2,7,5,3] target = 8 => true
# [2,7,5,3] target = 14 => false
def has_target(
    values, # List[int], unsorted & ranging between [-50000, 50000]
    target, # int
): # return a boolean
# for every number:
# - check to see if we're seen the compliment
# - If we have, we return true
# - If we have not, we will track the current nummber in a set and move on with our loop

  lookup = set()
  for val in values: 
    compliment = target - val
    if compliment in lookup: 
      return True
    
    lookup.add(val)
  
  return False


# Given a list of sorted (descending) integers and a target value:
# - Return true if 2 values at distinct indices sum up to target
# - Else return false
# Example:
# [7,5,3,2] target = 8 => true
# [7,5,3,2] target = 14 => false
def has_target_sorted(
    values, # List[int], sorted (descending) & ranging between [-50000, 50000]
    target, # int
):
  l = 0
  r = len(values) - 1

  while l < r: 
    if values[l] + values[r] == target: 
      return True
    
    if values[l] + values[r] < target: 
      r -= 1
    else: 
      l += 1
  
  return False

class TwoSum:

    
    def __init__(
        self,
        values,
    ):

      self.lookup = set()

      for i in range(len(values)): 
        for j in range(i + 1, len(values)):
          self.lookup.add(values[i] +  values[j])
      


    # NOTE: this method has to run faster than O(n) time
    #       for any possible invocation for every possible input for values.
    # - Return true if 2 values at distinct indices sum up to target
    # - Else return false
    def has_target(
        self,
        target
    ): # return a boolean
      return target in self.lookup
